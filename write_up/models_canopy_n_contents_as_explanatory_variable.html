data_complete_final <- 
  read.csv("/home/ecamo19/documents/projects/shade_house_exp/exploratory_figures_and_models/data/data_for_biomass_leaftraits_models.csv", header = T)


#Cleaned data
data_rate_change <- 
  
    data_complete_final %>% 
    
    #Order the columns
    dplyr::select(id,spcode,family,treatment,nfixer,init_height,
                  everything()) %>% 
    clean_names() %>% 
    mutate(biomass_no_leaves = totalbiom - whole_leaf_dry_weight )
#Load RGR data
data_rgr <- 
    read.csv("/home/ecamo19/documents/projects/shade_house_exp/exploratory_figures_and_models/data/finalDat.csv")

data_rgr_clean <- data_rgr %>% 
  dplyr::select(c(1,4, ncol(data_rgr)-1))
#Add RGR to the dataset
data_rate_change <- 
    
    #Join RGR  
    right_join(data_rate_change, data_rgr_clean, by = c("id", "spcode"))

2 Models

2.1 Model (m1): Total biomass as response variable

\[Total\ Biomass\sim treatment*fixer*Canopy\ nitrogen\ content + initial\ height\ + random(1|specie)\]

m1 <- lmer(totalbiom ~ treatment * nfixer * totalleafmass_nmass + init_height + 
             (1|spcode), data = data_rate_change)

Anova(m1, type = "III")
