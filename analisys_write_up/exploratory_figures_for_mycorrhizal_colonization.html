2 Mean data

2.1 Mycorrhizal colonization

summary_means <- 
    data_mycorrhizal_colonization_cleaned %>% 
    group_by(treatment,nfixer) %>%
    #log only the values greather than 0
    mutate(log_perc = if_else(percentage > 0, log(percentage),0)) %>% 
    summarise_if(is.numeric, funs(mean)) %>% 
    arrange(nfixer) 

### Round the numbers
summary_means[,5:ncol(summary_means)] <-
    round(summary_means[,5:ncol(summary_means)],3)

3 Boxplots: By fixer-non fixer

3.1 Mycorrhizal colonization

(boxplot_percentage <- 
    masomenos_boxplot(data = data_mycorrhizal_colonization_cleaned,
                      xvar = nfixer,
                      yvar = percentage, 
                      color = treatment,
                      n_treat = 4 ) +
    ylab("Mycorrhizal Colonization") )

(boxplot_percentage_log <- 
    masomenos_boxplot(data = data_mycorrhizal_colonization_cleaned,
                      xvar = nfixer,
                      yvar = log_perc, 
                      color = treatment,
                      n_treat = 4 ) +
    ylab("Mycorrhizal Colonization log(Percentage)") )
