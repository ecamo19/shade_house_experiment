5 Perform regression

Do PCs predict AGR?

# fit treatment/nfixer interactions separately
PCmod_agr_bothPC<- lmer(agr ~ treatment * PC1 +
                              nfixer * PC1 +
                              treatment * PC2 +
                              nfixer * PC2 +
                              init_height + (1|spcode), 
                        data = allTrait_scaled_forReg)
# fit three-way interaction
PCmod_agr_bothPC_3xItxn <- lmer(agr ~ treatment * nfixer* PC1+
                                      treatment * nfixer* PC2 +
                                      init_height + (1|spcode),
                                data = allTrait_scaled_forReg)

5.1 Model comparision

# check two-way vs. three-way interaction models (shows that two-way is way better, more negative AICc -- include in methods)
AICc(PCmod_agr_bothPC, PCmod_agr_bothPC_3xItxn)
                        df      AICc
PCmod_agr_bothPC        18 -271.4386
PCmod_agr_bothPC_3xItxn 27 -205.9787

5.2 Check Model assumptions

#check model assumptions with DHARMa -- could be worse... passes all tests so just go with it. 
simulationOutput <- simulateResiduals(fittedModel = PCmod_agr_bothPC)
plotQQunif(simulationOutput)
